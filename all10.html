        super.onCreate(b);
        setContentView(R.layout.activity_main);
        TextView t = findViewById(R.id.textView1);

            if (size > 40) size = 20;
            t.setTextSize(size += 4);

        findViewById(R.id.button2).setOnClickListener(v ->
            t.setTextColor(colors[i++ % 3]));
    }
}

package com.example.myapplication;
import android.os.Bundle;
import android.view.View;
import android.widget.*;

    EditText x, y; TextView r;

    protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_main);
        x = findViewById(R.id.editText1);
        y = findViewById(R.id.editText2);
        r = findViewById(R.id.ResultView);
    }

    public void calc(View v) {
        float a = Float.parseFloat(x.getText().toString());
        float b = Float.parseFloat(y.getText().toString());
        int id = v.getId();
        if (id == R.id.Add)       r.setText("" + (a+b));
        else if (id == R.id.Sub)  r.setText("" + (a-b));
        else if (id == R.id.Mul)  r.setText("" + (a*b));
        else if (id == R.id.Div)  r.setText("" + (a/b));
        else { x.setText(""); y.setText(""); r.setText(""); }
    }
}

package com.example.myapplication;
import android.app.Activity;
import android.graphics.*;
import android.os.Bundle;
import android.widget.ImageView;

public class MainActivity extends Activity {
    protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_main);

        Bitmap bm = Bitmap.createBitmap(800, 1400, Bitmap.Config.ARGB_8888);
        Canvas c = new Canvas(bm);
        Paint p = new Paint();

        // White background
        p.setColor(Color.WHITE);
        c.drawRect(0, 0, 800, 1400, p);

        // Draw shapes in MAGENTA
        p.setColor(Color.MAGENTA);
        p.setTextSize(50);

        c.drawText("Circle", 80, 80, p);
        c.drawCircle(200, 230, 130, p);

        c.drawText("Rectangle", 450, 80, p);
        c.drawRect(430, 100, 750, 380, p);

        c.drawText("Oval", 80, 480, p);
        c.drawOval(50, 500, 350, 750, p);

        c.drawText("Line", 500, 480, p);
        c.drawLine(580, 500, 580, 1100, p);

        ((ImageView) findViewById(R.id.imageView)).setImageBitmap(bm);
    }
}

HEAVY
☕ MainActivity.java
package com.example.myapplication;
import android.app.*;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.os.Bundle;
import android.view.View;
import android.widget.EditText;

public class MainActivity extends Activity implements View.OnClickListener {
    EditText Rollno, Name, Marks;
    SQLiteDatabase db;

    protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_main);
        Rollno = findViewById(R.id.Rollno);
        Name   = findViewById(R.id.Name);
        Marks  = findViewById(R.id.Marks);

        int[] ids = {R.id.Insert, R.id.Delete, R.id.Update,
                     R.id.View, R.id.ViewAll, R.id.Clear};
        for (int id : ids) findViewById(id).setOnClickListener(this);

        db = openOrCreateDatabase("StudentDB", MODE_PRIVATE, null);
        db.execSQL("CREATE TABLE IF NOT EXISTS student(" +
                   "rollno TEXT PRIMARY KEY, name TEXT, marks TEXT)");
    }

    public void onClick(View v) {
        String r = Rollno.getText().toString().trim();
        String n = Name.getText().toString().trim();
        String m = Marks.getText().toString().trim();
        int id = v.getId();

        if (id == R.id.Insert) {
            db.execSQL("INSERT OR REPLACE INTO student VALUES(?,?,?)",
                       new Object[]{r, n, m});
            show("Done", "Record Added"); clear();

        } else if (id == R.id.Delete) {
            db.execSQL("DELETE FROM student WHERE rollno=?", new Object[]{r});
            show("Done", "Deleted"); clear();

        } else if (id == R.id.Update) {
            db.execSQL("UPDATE student SET name=?, marks=? WHERE rollno=?",
                       new Object[]{n, m, r});
            show("Done", "Updated"); clear();

        } else if (id == R.id.View) {
            Cursor c = db.rawQuery(
                "SELECT * FROM student WHERE rollno=?", new String[]{r});
            if (c.moveToFirst()) {
                Name.setText(c.getString(1));
                Marks.setText(c.getString(2));
            } else show("Error", "Not Found");
            c.close();

        } else if (id == R.id.ViewAll) {
            Cursor c = db.rawQuery("SELECT * FROM student", null);
            StringBuilder sb = new StringBuilder();
            while (c.moveToNext())
                sb.append("R:").append(c.getString(0))
                  .append(" N:").append(c.getString(1))
                  .append(" M:").append(c.getString(2)).append("\n");
            c.close();
            show("All", sb.toString());

        } else clear();
    }

    void clear() { Rollno.setText(""); Name.setText(""); Marks.setText(""); }
    void show(String t, String msg) {
        new AlertDialog.Builder(this).setTitle(t).setMessage(msg)
            .setPositiveButton("OK", null).show();
    }
}

package com.example.myapplication;
import android.os.Bundle;
import android.view.View;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    TextView tv;

    protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_main);
        tv = findViewById(R.id.textView1);
    }

    public void fetchData(View v) {
        tv.setText("Started!\n");
        for (int i = 1; i <= 3; i++) {
            final int num = i;
            new Thread(() -> {
                for (int j = 0; j < 5; j++) {
                    try { Thread.sleep(1000); } catch (Exception e) {}
                    runOnUiThread(() -> tv.append("Thread " + num + " running\n"));
                }
            }).start();
        }
    }
}

GPS Location
MEDIUM
☕ MainActivity.java
package com.example.myapplication;
import android.Manifest;
import android.content.pm.PackageManager;
import android.location.LocationManager;
import android.os.Bundle;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_main);
        TextView tv = findViewById(R.id.tv);
        LocationManager lm = (LocationManager) getSystemService(LOCATION_SERVICE);

        if (checkSelfPermission(Manifest.permission.ACCESS_FINE_LOCATION)
                != PackageManager.PERMISSION_GRANTED) {
            requestPermissions(
                new String[]{Manifest.permission.ACCESS_FINE_LOCATION}, 1);
            return;
        }

        lm.requestLocationUpdates(LocationManager.GPS_PROVIDER, 500, 1,
            loc -> tv.setText(loc.getLatitude() + ", " + loc.getLongitude()));
    }
}
📄 AndroidManifest.xml — ADD THIS LINE before <application>

public class MainActivity extends AppCompatActivity {
    EditText e; TextView t; File f;

    protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_main);
        e = findViewById(R.id.i);
        t = findViewById(R.id.r);
        f = new File(getExternalFilesDir(null), "z.txt");
    }

    public void save(View v) throws Exception {
        Files.write(f.toPath(), e.getText().toString().getBytes());
        e.setText(""); t.setText("Saved");
    }

    public void read(View v) throws Exception {
        e.setText(new String(Files.readAllBytes(f.toPath())));
        t.setText("Read");
    }
}

Alarm Manager

package com.example.myapplication;
import android.app.*;
import android.content.*;
import android.os.Bundle;
import android.view.View;
import android.widget.*;
import java.util.Calendar;

public class MainActivity extends Activity {
    PendingIntent p;
    TimePicker t;

    protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_main);
        t = findViewById(R.id.timePicker);

        registerReceiver(new BroadcastReceiver() {
            public void onReceive(Context c, Intent i) {
                android.media.RingtoneManager.getRingtone(c,
                    android.media.RingtoneManager.getDefaultUri(4)).play();
            }
        }, new IntentFilter("A"), 2);
    }

    public void OnToggleClicked(View v) {
        AlarmManager m = (AlarmManager) getSystemService(ALARM_SERVICE);
        if (((ToggleButton) v).isChecked()) {
            Calendar c = Calendar.getInstance();
            c.set(11, t.getHour());
            c.set(12, t.getMinute());
            c.set(13, 0);
            p = PendingIntent.getBroadcast(this, 0, new Intent("A"), 1<<26);
            m.set(0, c.getTimeInMillis(), p);
            Toast.makeText(this, "Alarm Set", 0).show();
        } else {
            if (p != null) m.cancel(p);
        }
    }
}
